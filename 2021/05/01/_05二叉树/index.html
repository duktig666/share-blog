# дәҢеҸүж ‘зҡ„йҮҚиҰҒжҖ§

е…ҲеҲ·вјҶеҸүж ‘зҡ„йўҳвҪ¬пјҢе…ҲеҲ·вјҶеҸүж ‘зҡ„йўҳвҪ¬пјҢе…ҲеҲ·вјҶеҸүж ‘зҡ„йўҳвҪ¬пјҢеӣ дёәеҫҲеӨҡз»Ҹе…ёз®—жі•пјҢд»ҘеҸҠжҲ‘д»¬еүҚвҪӮи®ІиҝҮзҡ„жүҖжңүеӣһжәҜгҖҒеҠЁеҪ’гҖҒеҲҶжІ»з®—жі•пјҢе…¶е®һйғҪжҳҜж ‘зҡ„й—®йўҳпјҢвҪҪж ‘зҡ„й—®йўҳе°ұж°ёиҝңйҖғдёҚејҖж ‘зҡ„йҖ’еҪ’йҒҚеҺҶжЎҶжһ¶иҝҷвјҸвҫҸз ҙд»Јз Ғпјҡ

еҝ«йҖҹжҺ’еәҸзҡ„йҖ»иҫ‘жҳҜпјҢиӢҘиҰҒеҜ№ nums[lo..hi] иҝӣвҫҸжҺ’еәҸпјҢжҲ‘д»¬е…ҲжүҫвјҖдёӘеҲҶз•ҢзӮ№ pпјҢйҖҡиҝҮдәӨжҚўе…ғзҙ дҪҝеҫ—nums[lo..p-1] йғҪвј©дәҺзӯүдәҺ nums[p]пјҢдё” nums[p+1..hi] йғҪвјӨдәҺnums[p]пјҢз„¶еҗҺйҖ’еҪ’ең°еҺ»nums[lo..p-1] е’Ң nums[p+1..hi] дёӯеҜ»жүҫж–°зҡ„еҲҶз•ҢзӮ№пјҢжңҖеҗҺж•ҙдёӘж•°з»„е°ұиў«жҺ’еәҸдәҶгҖӮ

еҰӮжһңдҪ вјҖзңје°ұиҜҶз ҙиҝҷдәӣжҺ’еәҸз®—жі•зҡ„еә•з»ҶпјҢиҝҳйңҖиҰҒиғҢиҝҷдәӣз®—жі•д»Јз Ғеҗ—пјҹиҝҷдёҚжҳҜвјҝеҲ°ж“’жқҘпјҢд»ҺжЎҶжһ¶ж…ўж…ўжү©еұ•е°ұиғҪеҶҷеҮәз®—жі•дәҶгҖӮ

ж—ЁеңЁиҜҙжҳҺпјҢвјҶеҸүж ‘зҡ„з®—жі•жҖқжғізҡ„иҝҗвҪӨвјҙжіӣпјҢз”ҡвҫ„еҸҜд»ҘиҜҙпјҢеҸӘиҰҒж¶үеҸҠйҖ’еҪ’пјҢйғҪеҸҜд»ҘжҠҪиұЎжҲҗвјҶеҸүж ‘зҡ„й—®йўҳгҖӮ

# еҶҷйҖ’еҪ’з®—жі•зҡ„з§ҳиҜҖ

еҶҷйҖ’еҪ’з®—жі•зҡ„е…ій”®жҳҜиҰҒжҳҺзЎ®еҮҪж•°зҡ„гҖҢе®ҡд№үгҖҚжҳҜд»Җд№ҲпјҢз„¶еҗҺзӣёдҝЎиҝҷдёӘе®ҡд№үпјҢеҲ©вҪӨиҝҷдёӘе®ҡд№үжҺЁеҜјжңҖз»Ҳз»“жһңпјҢз»қдёҚиҰҒи·івјҠйҖ’еҪ’зҡ„з»ҶиҠӮгҖӮ

вҪӨвјҖдёӘе…·дҪ“зҡ„дҫӢвјҰжқҘиҜҙпјҢвҪҗеҰӮиҜҙи®©дҪ и®Ўз®—вјҖжЈөвјҶеҸүж ‘е…ұжңүвјҸдёӘиҠӮзӮ№пјҡ

// е®ҡд№үпјҡcount(root) иҝ”еӣһд»Ҙ root дёәж №зҡ„ж ‘жңүеӨҡе°‘иҠӮзӮ№
int count(TreeNode root) { 
    // base case 
    if (root == null) return 0; 
    // вҫғвј°еҠ дёҠвјҰж ‘зҡ„иҠӮзӮ№ж•°е°ұжҳҜж•ҙжЈөж ‘зҡ„иҠӮзӮ№ж•° 
    return 1 + count(root.left) + count(root.right); 
}

иҝҷдёӘй—®йўҳвҫ®еёёз®ҖеҚ•пјҢвјӨе®¶еә”иҜҘйғҪдјҡеҶҷиҝҷж®өд»Јз ҒпјҢroot жң¬иә«е°ұжҳҜвјҖдёӘиҠӮзӮ№пјҢеҠ дёҠе·ҰеҸівјҰж ‘зҡ„иҠӮзӮ№ж•°е°ұжҳҜд»Ҙroot дёәж №зҡ„ж ‘зҡ„иҠӮзӮ№жҖ»ж•°гҖӮ

е·ҰеҸівјҰж ‘зҡ„иҠӮзӮ№ж•°жҖҺд№Ҳз®—пјҹе…¶е®һе°ұжҳҜи®Ўз®—ж №дёә root.left е’Ң root.right дёӨжЈөж ‘зҡ„иҠӮзӮ№ж•°е‘—пјҢжҢүз…§е®ҡд№үпјҢйҖ’еҪ’и°ғвҪӨ count еҮҪж•°еҚіеҸҜз®—еҮәжқҘгҖӮ

еҶҷж ‘зӣёе…ізҡ„з®—жі•пјҢз®ҖеҚ•иҜҙе°ұжҳҜпјҢе…Ҳжҗһжё…жҘҡеҪ“еүҚ root иҠӮзӮ№гҖҢиҜҘеҒҡд»Җд№ҲгҖҚд»ҘеҸҠгҖҢд»Җд№Ҳж—¶еҖҷеҒҡгҖҚпјҢз„¶еҗҺж №жҚ®еҮҪж•°е®ҡд№үйҖ’еҪ’и°ғвҪӨвјҰиҠӮзӮ№пјҢйҖ’еҪ’и°ғвҪӨдјҡи®©еӯ©вјҰиҠӮзӮ№еҒҡзӣёеҗҢзҡ„дәӢжғ…гҖӮ

жүҖи°“гҖҢиҜҘеҒҡд»Җд№ҲгҖҚе°ұжҳҜи®©дҪ жғіжё…жҘҡеҶҷд»Җд№Ҳд»Јз ҒиғҪеӨҹе®һзҺ°йўҳвҪ¬жғіиҰҒзҡ„ж•ҲжһңпјҢжүҖи°“гҖҢд»Җд№Ҳж—¶еҖҷеҒҡгҖҚпјҢе°ұжҳҜи®©дҪ жҖқиҖғиҝҷж®өд»Јз ҒеҲ°еә•еә”иҜҘеҶҷеңЁеүҚеәҸгҖҒдёӯеәҸиҝҳжҳҜеҗҺеәҸйҒҚеҺҶзҡ„д»Јз ҒдҪҚзҪ®дёҠгҖӮ

TreeNode invertTree(TreeNode root) { 
    // base case 
    if (root == null) { 
        return null; 
    } 
 
    /**** еүҚеәҸйҒҚеҺҶдҪҚзҪ® ****/ 
    // root иҠӮзӮ№йңҖиҰҒдәӨжҚўе®ғзҡ„е·ҰеҸівјҰиҠӮзӮ№

    invertTree(root.right); 
     
    return root; 
}

// е®ҡд№үпјҡе°Ҷд»Ҙ root дёәж №зҡ„ж ‘жӢүе№ідёәй“ҫиЎЁ
void flatten(TreeNode root) { 
    // base case 
    if (root == null) return; 
     
    flatten(root.left); 
    flatten(root.right); 
 
    /* еҗҺеәҸйҒҚеҺҶдҪҚзҪ® */ 
    // 1гҖҒе·ҰеҸівјҰж ‘е·Із»Ҹиў«жӢүе№іжҲҗвјҖжқЎй“ҫиЎЁ 
    TreeNode left = root.left; 
    TreeNode right = root.right; 
     
    // 2гҖҒе°Ҷе·ҰвјҰж ‘дҪңдёәеҸівјҰж ‘ 
    root.left = null; 
    root.right = left; 
 
    // 3гҖҒе°ҶеҺҹе…Ҳзҡ„еҸівјҰж ‘жҺҘеҲ°еҪ“еүҚеҸівјҰж ‘зҡ„жң«з«Ҝ 
    TreeNode p = root; 
    while (p.right != null) { 
        p = p.right; 
    } 
    p.right = right; 
}

public TreeNode constructMaximumBinaryTree(int[] nums) {
    return build(nums, 0, nums.length - 1);

/**
 * е°Ҷ nums[low..high] жһ„йҖ жҲҗз¬ҰеҗҲжқЎд»¶зҡ„ж ‘пјҢиҝ”еӣһж №иҠӮзӮ№
 *
 * @param nums еҫ…жһ„е»әдәҢеҸүж ‘зҡ„ж•°з»„
 * @param low  еӯҗж ‘зҡ„е·Ұз«Ҝж•°з»„дёӢж Ү
 * @param high еӯҗж ‘зҡ„еҸіз«Ҝж•°з»„дёӢж Ү
 * @return жһ„йҖ еҘҪеӯҗж ‘зҡ„ж №иҠӮзӮ№
 */
private TreeNode build(int[] nums, int low, int high) {
    // йҖ’еҪ’зҡ„з»ҲжӯўжқЎд»¶
    if (low > high) {
        return null;
    }

    // жүҫеҲ°ж•°з»„дёӯзҡ„жңҖеӨ§еҖје’ҢеҜ№еә”зҡ„зҙўеј•
    int index = - 1, maxVal = Integer.MIN_VALUE;
    for (int i = low; i <= high; i++) {
        if (maxVal < nums[i]) {
            maxVal = nums[i];
            index = i;
        }
    }

    // жңҖеӨ§еҖјжһ„е»әж №иҠӮзӮ№
    TreeNode root = new TreeNode(maxVal);
    // йҖ’еҪ’жһ„е»әе·ҰеҸіеӯҗж ‘
    root.left = build(nums, low, index - 1);
    root.right = build(nums, index + 1, high);

    return root;
}

public TreeNode buildTree(int[] preorder, int[] inorder) {

/**
 * иӢҘеүҚеәҸйҒҚеҺҶж•°з»„дёә preorder[preStart..preEnd]пјҢ
 * дёӯеәҸйҒҚеҺҶж•°з»„дёә inorder[inStart..inEnd]пјҢ
 * жһ„йҖ вјҶеҸүж ‘пјҢиҝ”еӣһиҜҘвјҶеҸүж ‘зҡ„ж №иҠӮзӮ№
 */
private TreeNode build(int[] preorder, int preStart, int preEnd,
                       int[] inorder, int inStart, int inEnd) {
    // йҖ’еҪ’з»ҲжӯўжқЎд»¶
    if (preStart > preEnd) {
        return null;
    }

    // root иҠӮзӮ№еҜ№еә”зҡ„еҖје°ұжҳҜеүҚеәҸйҒҚеҺҶж•°з»„зҡ„з¬¬вјҖдёӘе…ғзҙ 
    int rootVal = preorder[preStart];
    // rootVal еңЁдёӯеәҸйҒҚеҺҶж•°з»„дёӯзҡ„зҙўеј•
    int index = 0;
    for (int i = inStart; i <= inEnd; i++) {
        if (inorder[i] == rootVal) {
            index = i;
            break;
        }
    }

    // ж №иҠӮзӮ№е·Ұиҫ№жңүеҮ дёӘе…ғзҙ 
    int leftSize = index - inStart;

    // е…Ҳжһ„йҖ еҮәеҪ“еүҚж №иҠӮзӮ№
    TreeNode root = new TreeNode(rootVal);

    // йҖ’еҪ’жһ„йҖ е·ҰеҸівјҰж ‘
    root.left = build(preorder, preStart + 1, preStart + leftSize,
            inorder, inStart, index - 1);

    root.right = build(preorder, preStart + leftSize + 1, preEnd,
            inorder, index + 1, inEnd);

    return root;
}

106. д»ҺдёӯеәҸдёҺеҗҺеәҸйҒҚеҺҶеәҸеҲ—жһ„йҖ дәҢеҸүж ‘ (opens new window)

ж №жҚ®дёҖжЈөж ‘зҡ„дёӯеәҸйҒҚеҺҶдёҺеҗҺеәҸйҒҚеҺҶжһ„йҖ дәҢеҸүж ‘гҖӮ

иҝҷйҒ“йўҳе’ҢдёҠвјҖйўҳзҡ„е…ій”®еҢәеҲ«жҳҜпјҢеҗҺеәҸйҒҚеҺҶе’ҢеүҚеәҸйҒҚеҺҶзӣёеҸҚпјҢж №иҠӮзӮ№еҜ№еә”зҡ„еҖјдёә postorder зҡ„жңҖеҗҺвјҖдёӘе…ғзҙ гҖӮ

/**
 * дё»еҮҪж•°
 */
public TreeNode buildTree(int[] inorder, int[] postorder) {
    return build(inorder, 0, inorder.length - 1,
            postorder, 0, postorder.length - 1);
}

/**
 * дёӯеәҸйҒҚеҺҶж•°з»„дёә inorder[inStart..inEnd]пјҢ
 * еҗҺеәҸйҒҚеҺҶж•°з»„дёә postorder[postStart..postEnd]пјҢ
 * жһ„йҖ вјҶеҸүж ‘пјҢиҝ”еӣһиҜҘвјҶеҸүж ‘зҡ„ж №иҠӮзӮ№
 */
private TreeNode build(int[] inorder, int inStart, int inEnd,
                       int[] postorder, int postStart, int postEnd) {
    // йҖ’еҪ’з»ҲжӯўжқЎд»¶
    if (inStart > inEnd) {
        return null;
    }
    
    // root иҠӮзӮ№еҜ№еә”зҡ„еҖје°ұжҳҜеҗҺеәҸйҒҚеҺҶж•°з»„зҡ„жңҖеҗҺвјҖдёӘе…ғзҙ 
    int rootVal = postorder[postEnd];
    
    // rootVal еңЁдёӯеәҸйҒҚеҺҶж•°з»„дёӯзҡ„зҙўеј•
    int index = 0;
    for (int i = inStart; i <= inEnd; i++) {
        if (inorder[i] == rootVal) {
            index = i;
            break;
        }
    }
    
    // е·ҰвјҰж ‘зҡ„иҠӮзӮ№дёӘж•°
    int leftSize = index - inStart;
    TreeNode root = new TreeNode(rootVal);
    // йҖ’еҪ’жһ„йҖ е·ҰеҸівјҰж ‘
    root.left = build(inorder, inStart, index - 1,
            postorder, postStart, postStart + leftSize - 1);

    root.right = build(inorder, index + 1, inEnd,
            postorder, postStart + leftSize, postEnd - 1);
    
    return root;
}

    return res;
}

е…¶е®һиҝҷе°ұжҳҜжҠҠд№ӢеүҚеҲҶжһҗдёӯиҜҙеҲ°зҡ„еҮ дёӘеҖјж”ҫеҲ°дәҶ res ж•°з»„дёӯпјҢжңҖйҮҚиҰҒзҡ„жҳҜпјҢжҲ‘д»¬иҰҒиҜ•еӣҫйҖҡиҝҮ left е’Ң right жӯЈзЎ®жҺЁеҜјеҮә res ж•°з»„гҖӮ

/** е…ЁеұҖеҸҳйҮҸпјҢи®°еҪ• BST жңҖеӨ§иҠӮзӮ№д№Ӣе’Ң */
int maxSum = 0;

/** дё»еҮҪж•° */
public int maxSumBST(TreeNode root) {
    traverse(root);
    return maxSum;
}

/**
 * йҖ’еҪ’и®Ўз®—
 *
 * @param root /
 * @return int[]{ isBST, min, max, sum}
 */
private int[] traverse(TreeNode root) {
    // base case
    if (root == null) {
        return new int[] {
                1, Integer.MAX_VALUE, Integer.MIN_VALUE, 0
        };
    }

    // йҖ’еҪ’и®Ўз®—е·ҰеҸіеӯҗж ‘
    int[] left = traverse(root.left);
    int[] right = traverse(root.right);

    /*  еҗҺеәҸйҒҚеҺҶдҪҚзҪ®  */
    int[] res = new int[4];
    // иҝҷдёӘ if еңЁеҲӨж–ӯд»Ҙ root дёәж №зҡ„дәҢеҸүж ‘жҳҜдёҚжҳҜ BST
    if (left[0] == 1 && right[0] == 1 &&
            root.val > left[2] && root.val < right[1]) {
        // д»Ҙ root дёәж №зҡ„дәҢеҸүж ‘жҳҜ BST
        res[0] = 1;
        // и®Ўз®—д»Ҙ root дёәж №зҡ„иҝҷжЈө BST зҡ„жңҖе°ҸеҖј
        res[1] = Math.min(left[1], root.val);
        // и®Ўз®—д»Ҙ root дёәж №зҡ„иҝҷжЈө BST зҡ„жңҖеӨ§еҖј
        res[2] = Math.max(right[2], root.val);
        // и®Ўз®—д»Ҙ root дёәж №зҡ„иҝҷжЈө BST жүҖжңүиҠӮзӮ№д№Ӣе’Ң
        res[3] = left[3] + right[3] + root.val;
        // жӣҙж–°е…ЁеұҖеҸҳйҮҸ
        maxSum = Math.max(maxSum, res[3]);
    } else {
        // д»Ҙ root дёәж №зҡ„дәҢеҸүж ‘дёҚжҳҜ BST
        res[0] = 0;
        // е…¶д»–зҡ„еҖјйғҪжІЎеҝ…иҰҒи®Ўз®—дәҶпјҢеӣ дёәз”ЁдёҚеҲ°
    }
    return res;
}

TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
    // base case
    if (root == null) return null;
    if (root == p || root == q) return root;

    TreeNode left = lowestCommonAncestor(root.left, p, q);
    TreeNode right = lowestCommonAncestor(root.right, p, q);
    // жғ…еҶө 1
    if (left != null && right != null) {
        return root;
    }
    // жғ…еҶө 2
    if (left == null && right == null) {
        return null;
    }
    // жғ…еҶө 3
    return left == null ? right : left;
}

    if (k == rank) {

        return;

    traverse(root.right, k);
}

            && isValidBST(root.right, root, max);

/* и®Ўз®—й—ӯеҢәй—ҙ [lo, hi] з»„жҲҗзҡ„ BST дёӘж•° */
int count(int lo, int hi) {
    // base case
    if (lo > hi) return 1;

    int res = 0;
    for (int i = lo; i <= hi; i++) {
        // i зҡ„еҖјдҪңдёәж №иҠӮзӮ№ root
        int left = count(lo, i - 1);
        int right = count(i + 1, hi);

        res += left * right;
    }

    return res;
}

/** еӨҮеҝҳеҪ• */
private int[][] memo;

/**
 * 96йўҳ
 */
public int numTrees(int n) {
    // еӨҮеҝҳеҪ•зҡ„еҖјеҲқе§ӢеҢ–дёә 0
    memo = new int[n + 1][n + 1];
    // и®Ўз®—й—ӯеҢәй—ҙ [1, n] з»„жҲҗзҡ„ BST дёӘж•°
    return count(1, n);
}

/** и®Ўз®—й—ӯеҢәй—ҙ [lo, hi] з»„жҲҗзҡ„ BST дёӘж•° */
private int count(int lo, int hi) {
    // base case з©әиҠӮзӮ№пјҢд№ҹжҳҜдёҖз§Қжғ…еҶөпјҢиҝ”еӣһ 1
    if (lo > hi) {
        return 1;
    }

    // жҹҘеӨҮеҝҳеҪ•
    if (memo[lo][hi] != 0) {
        return memo[lo][hi];
    }

    int res = 0;
    for (int i = lo; i <= hi; i++) {
        // i зҡ„еҖјдҪңдёәж №иҠӮзӮ№ root
        int left = count(lo, i - 1);
        int right = count(i + 1, hi);
        // е·ҰеҸіеӯҗж ‘зҡ„з»„еҗҲж•°д№ҳз§ҜжҳҜ BST зҡ„жҖ»ж•°
        res += left * right;
    }

    // е°Ҷз»“жһңеӯҳе…ҘеӨҮеҝҳеҪ•
    memo[lo][hi] = res;

    return res;
}

/**
 * 95йўҳ
 */
public List<TreeNode> generateTrees(int n) {
    if (n == 0) {
        return new LinkedList<>();
    }
    // жһ„йҖ й—ӯеҢәй—ҙ [1, n] з»„жҲҗзҡ„ BST
    return build(1, n);
}

/** жһ„йҖ й—ӯеҢәй—ҙ [lo, hi] з»„жҲҗзҡ„ BST */
private List<TreeNode> build(int lo, int hi) {
    List<TreeNode> res = new LinkedList<>();
    // base case
    if (lo > hi) {
        res.add(null);
        return res;
    }

    // 1гҖҒз©·дёҫ root иҠӮзӮ№зҡ„жүҖжңүеҸҜиғҪгҖӮ
    for (int i = lo; i <= hi; i++) {
        // 2гҖҒйҖ’еҪ’жһ„йҖ еҮәе·ҰеҸіеӯҗж ‘зҡ„жүҖжңүеҗҲжі• BSTгҖӮ
        List<TreeNode> leftTree = build(lo, i - 1);
        List<TreeNode> rightTree = build(i + 1, hi);
        // 3гҖҒз»ҷ root иҠӮзӮ№з©·дёҫжүҖжңүе·ҰеҸіеӯҗж ‘зҡ„з»„еҗҲгҖӮ
        for (TreeNode left : leftTree) {
            for (TreeNode right : rightTree) {
                // i дҪңдёәж №иҠӮзӮ№ root зҡ„еҖј
                TreeNode root = new TreeNode(i);
                root.left = left;
                root.right = right;
                res.add(root);
            }
        }
    }

    return res;
}
